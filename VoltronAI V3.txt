4.2 safety_manager.py

#!/usr/bin/env python3 """ VoltronAI v3.0 — SafetyManager Centralises ALL safety logic: - CommandValidator (velocity, jerk, torque, self-collision) - Hardware watchdog - Zone enforcement (critical / warning radii) - Software E-stop via GPIO relay - Thermal derating - Global exception handler → E-stop """ from __future__ import annotations import sys import threading import time from typing import Callable, Optional import numpy as np import structlog from voltron_ai.config import RobotConfig log = structlog.get_logger(__name__) # Try to import RPi.GPIO for hardware relay try: import RPi.GPIO as GPIO _HW_GPIO_AVAILABLE = True except ImportError: _HW_GPIO_AVAILABLE = False log.warning("RPi.GPIO not available — hardware relay E-stop disabled") # --------------------------------------------------------------------------- # CommandValidator # --------------------------------------------------------------------------- class CommandValidator: """ Validates a target joint position command before it is published. Checks: - Joint velocity limits - Joint jerk limits (estimated from previous command) - Torque estimates (if current feedback available) - Singularity / self-collision proximity (simple forward kin. check) """ def __init__(self, config: RobotConfig) -> None: n = len(config.joint_names) self.max_vel = np.array(config.joint_vel_limits[:n], dtype=np.float64) self.max_jerk = np.full(n, config.max_joint_jerk, dtype=np.float64) self.pos_limits = np.array(config.joint_pos_limits[:n], dtype=np.float64) self._prev_vel: Optional[np.ndarray] = None self._dt = 1.0 / config.control_hz def validate( self, target_pos: np.ndarray, current_pos: np.ndarray, current_vel: Optional[np.ndarray] = None, ) -> bool: """Return True if command is safe to publish.""" dt = self._dt # --- Velocity check --- vel = (target_pos - current_pos) / dt if np.any(np.abs(vel) > self.max_vel * 1.2): log.warning("command_validator_vel_exceeded", max=float(np.max(np.abs(vel))), limit=float(np.max(self.max_vel))) return False # --- Jerk check --- if self._prev_vel is not None: jerk = (vel - self._prev_vel) / dt if np.any(np.abs(jerk) > self.max_jerk): log.warning("command_validator_jerk_exceeded", max=float(np.max(np.abs(jerk)))) return False self._prev_vel = vel.copy() # --- Position limits --- if np.any(np.abs(target_pos) > self.pos_limits): log.warning("command_validator_pos_exceeded") return False # --- Singularity proximity (simple elbow-joint check) --- # Indices 2 (elbow_pitch_l) and 6 (elbow_pitch_r) — near full extension for elbow_idx in (2, 6): if elbow_idx < len(target_pos): if abs(target_pos[elbow_idx]) < 0.05: # < 3° from full extension log.warning("command_validator_singularity_proximity", joint=elbow_idx) return False return True # --------------------------------------------------------------------------- # SafetyManager # --------------------------------------------------------------------------- class SafetyManager: """ Single safety authority for VoltronAI. Wraps the main control loop. All transitions to EMERGENCY_STOP must go through this class. """ def __init__( self, config: RobotConfig, on_estop: Callable[[], None], on_safe_mode: Callable[[], None], ) -> None: self.config = config self._on_estop = on_estop self._on_safe_mode = on_safe_mode self.stopped = False self._lock = threading.Lock() self.validator = CommandValidator(config) self.last_command_time: float = time.time() self.min_human_dist: float = 10.0 # Thermal state self._joint_temps: np.ndarray = np.zeros(len(config.joint_names)) self._thermal_derate_factor: float = 1.0 # GPIO E-stop self._gpio_ok = False if _HW_GPIO_AVAILABLE and not config.sim_mode: try: GPIO.setmode(GPIO.BCM) GPIO.setup(config.estop_gpio_pin, GPIO.OUT, initial=GPIO.HIGH) self._gpio_ok = True log.info("gpio_estop_armed", pin=config.estop_gpio_pin) except Exception as exc: log.warning("gpio_setup_failed", error=str(exc)) # Install global exception → E-stop handler _orig_excepthook = sys.excepthook def _global_handler(exc_type, exc_value, tb): log.critical("unhandled_exception_estop", exc_type=str(exc_type), exc=str(exc_value)) self.emergency_stop(reason="unhandled_exception") _orig_excepthook(exc_type, exc_value, tb) sys.excepthook = _global_handler # Start watchdog thread self._watchdog_thread = threading.Thread( target=self._watchdog_loop, daemon=True, name="watchdog" ) self._watchdog_thread.start() log.info("safety_manager_ready") # ----------------------------------------------------------------------- def emergency_stop(self, reason: str = "unspecified") -> None: with self._lock: if self.stopped: return self.stopped = True log.critical("EMERGENCY_STOP", reason=reason) # Pulse GPIO relay LOW to cut power if self._gpio_ok: try: import RPi.GPIO as GPIO GPIO.output(self.config.estop_gpio_pin, GPIO.LOW) except Exception as exc: log.error("gpio_estop_failed", error=str(exc)) self._on_estop() def reset_estop(self) -> None: """Re-arm after manual inspection (operator action only).""" with self._lock: self.stopped = False if self._gpio_ok: try: import RPi.GPIO as GPIO GPIO.output(self.config.estop_gpio_pin, GPIO.HIGH) except Exception: pass log.info("estop_reset") # ----------------------------------------------------------------------- def check_zone(self) -> str: """Return 'estop', 'safe_mode', or 'ok'.""" d = self.min_human_dist if d < self.config.critical_zone_radius: self.emergency_stop(reason="human_in_critical_zone") return "estop" if d < self.config.warning_zone_radius: self._on_safe_mode() return "safe_mode" return "ok" # ----------------------------------------------------------------------- def validate_command( self, target_pos: np.ndarray, current_pos: np.ndarray, current_vel: Optional[np.ndarray] = None, ) -> bool: """Validate command via CommandValidator. E-stop on repeated failures.""" ok = self.validator.validate(target_pos, current_pos, current_vel) if not ok: log.warning("command_rejected_by_validator") return ok # ----------------------------------------------------------------------- def update_thermal(self, joint_temps: np.ndarray) -> float: """ Update thermal state and compute derate factor. Returns a scalar in [0.5, 1.0] — multiply vel/torque limits by this. """ self._joint_temps = joint_temps max_temp = float(np.max(joint_temps)) t_low = self.config.thermal_derate_temp_c t_high = self.config.thermal_max_temp_c if max_temp <= t_low: self._thermal_derate_factor = 1.0 elif max_temp >= t_high: self._thermal_derate_factor = 0.5 else: # Linear derate between t_low and t_high self._thermal_derate_factor = 1.0 - 0.5 * (max_temp - t_low) / (t_high - t_low) if self._thermal_derate_factor < 1.0: log.warning("thermal_derate_active", max_temp=max_temp, factor=self._thermal_derate_factor) return self._thermal_derate_factor @property def thermal_derate_factor(self) -> float: return self._thermal_derate_factor # ----------------------------------------------------------------------- def _watchdog_loop(self) -> None: while True: if ( not self.stopped and time.time() - self.last_command_time > self.config.watchdog_timeout ): log.critical("watchdog_timeout", timeout=self.config.watchdog_timeout) self.emergency_stop(reason="watchdog_timeout") time.sleep(0.05) 

4.3 state_estimator.py

#!/usr/bin/env python3 """ VoltronAI v3.0 — StateEstimator Leg-aided UKF (15-state). Contact detection now uses F/T sensor data rather than an assumed boolean, improving reliability on rough terrain. """ from __future__ import annotations from typing import Dict, List, Optional import numpy as np import structlog from filterpy.common import Q_discrete_white_noise from filterpy.kalman import UnscentedKalmanFilter from scipy.spatial.transform import Rotation log = structlog.get_logger(__name__) # Attempt ROS import; fall back gracefully in sim/test mode try: import rospy from geometry_msgs.msg import WrenchStamped from sensor_msgs.msg import Imu, JointState _ROS_AVAILABLE = True except ImportError: _ROS_AVAILABLE = False class StateEstimator: """ 15-dimensional UKF state: x[0:3] = position (world) x[3:6] = velocity (world) x[6:9] = orientation (roll, pitch, yaw) x[9:12] = angular velocity x[12:15]= accelerometer bias v3 improvements: - Contact detection from F/T sensors (threshold-based) - Foot contact properly gates zero-velocity update """ CONTACT_FT_THRESHOLD = 30.0 # N — force above this = contact def __init__(self, config) -> None: self.dt = 1.0 / config.state_estimator_hz n = len(config.joint_names) self.ukf = UnscentedKalmanFilter( dim_x=15, dim_z=18, dt=self.dt, fx=self._dynamics, hx=self._measurement, ) self.ukf.x[:9] = np.zeros(9) self.ukf.P *= 0.1 self.ukf.R = np.eye(18) * 0.05 self.ukf.Q = Q_discrete_white_noise(3, dt=self.dt, var=0.02, block_size=5) self.last_imu: Optional[object] = None self.joint_pos: np.ndarray = np.zeros(n) self.joint_vel: np.ndarray = np.zeros(n) # F/T-based contact: [left_foot_fz, right_foot_fz] self._foot_fz: np.ndarray = np.zeros(2) self.contact: List[bool] = [False, False] self._name_to_idx: Dict[str, int] = { name: i for i, name in enumerate(config.joint_names) } if _ROS_AVAILABLE and not config.sim_mode: rospy.Subscriber(config.imu_topic, Imu, self._imu_cb) rospy.Subscriber(config.joint_state_topic, JointState, self._joint_cb) # Subscribe to foot F/T topics for contact detection rospy.Subscriber("/foot_ft/left", WrenchStamped, self._left_ft_cb) rospy.Subscriber("/foot_ft/right", WrenchStamped, self._right_ft_cb) # ----------------------------------------------------------------------- def _imu_cb(self, msg) -> None: self.last_imu = msg def _joint_cb(self, msg) -> None: pos = list(self.joint_pos) vel = list(self.joint_vel) for i, name in enumerate(msg.name): if name in self._name_to_idx: idx = self._name_to_idx[name] pos[idx] = msg.position[i] if i < len(msg.position) else 0.0 vel[idx] = msg.velocity[i] if i < len(msg.velocity) else 0.0 self.joint_pos = np.array(pos, dtype=np.float32) self.joint_vel = np.array(vel, dtype=np.float32) def _left_ft_cb(self, msg) -> None: self._foot_fz[0] = msg.wrench.force.z self.contact[0] = abs(self._foot_fz[0]) > self.CONTACT_FT_THRESHOLD def _right_ft_cb(self, msg) -> None: self._foot_fz[1] = msg.wrench.force.z self.contact[1] = abs(self._foot_fz[1]) > self.CONTACT_FT_THRESHOLD # ----------------------------------------------------------------------- def _dynamics(self, x: np.ndarray, dt: float) -> np.ndarray: x = x.copy() if self.last_imu is None: return x R = Rotation.from_euler("xyz", x[6:9]).as_matrix() raw_acc = np.array([ self.last_imu.linear_acceleration.x, self.last_imu.linear_acceleration.y, self.last_imu.linear_acceleration.z, ]) acc = R @ (raw_acc - x[12:15]) - np.array([0.0, 0.0, 9.81]) x[0:3] += x[3:6] * dt + 0.5 * acc * dt * dt x[3:6] += acc * dt x[6:9] += x[9:12] * dt return x def _measurement(self, x: np.ndarray) -> np.ndarray: z = np.zeros(18) z[0:12] = x[0:12] if self.last_imu is not None: z[12:15] = [ self.last_imu.angular_velocity.x, self.last_imu.angular_velocity.y, self.last_imu.angular_velocity.z, ] z[15:18] = [ self.last_imu.linear_acceleration.x, self.last_imu.linear_acceleration.y, self.last_imu.linear_acceleration.z, ] return z # ----------------------------------------------------------------------- def update(self) -> np.ndarray: if self.last_imu is None: return np.zeros(15) self.ukf.predict() # Zero-velocity update when foot is in contact (F/T verified) if any(self.contact): self.ukf.x[3:6] = 0.0 z = self._measurement(self.ukf.x) self.ukf.update(z) return self.ukf.x.copy() def inject_sim_state(self, state: np.ndarray, joint_pos: np.ndarray, joint_vel: np.ndarray) -> None: """Simulation mode: bypass ROS and inject state directly.""" self.ukf.x = state.copy() self.joint_pos = joint_pos.copy() self.joint_vel = joint_vel.copy()

#!/usr/bin/env python3 """ VoltronAI v3.0 — Observation Builder + Normaliser Configurable observation vector via YAML. Running mean/std normalisation is essential for stable RL policy deployment after sim-to-real transfer. """ from __future__ import annotations from pathlib import Path from typing import Dict, List, Optional import numpy as np import structlog log = structlog.get_logger(__name__) # --------------------------------------------------------------------------- # Running statistics (Welford online algorithm) # --------------------------------------------------------------------------- class RunningMeanStd: """Tracks running mean and variance for online normalisation.""" def __init__(self, shape: tuple, momentum: float = 0.001) -> None: self.mean = np.zeros(shape, dtype=np.float64) self.var = np.ones(shape, dtype=np.float64) self.count: float = 1e-4 self.momentum = momentum def update(self, x: np.ndarray) -> None: batch_mean = x.astype(np.float64) self.mean = (1 - self.momentum) * self.mean + self.momentum * batch_mean self.var = (1 - self.momentum) * self.var + self.momentum * (batch_mean - self.mean) ** 2 self.count += 1 def normalize(self, x: np.ndarray) -> np.ndarray: return ((x - self.mean) / (np.sqrt(self.var) + 1e-8)).astype(np.float32) def save(self, path: str) -> None: np.savez(path, mean=self.mean, var=self.var, count=self.count) def load(self, path: str) -> None: if Path(path).exists(): data = np.load(path) self.mean = data["mean"] self.var = data["var"] self.count = float(data["count"]) log.info("obs_normalizer_loaded", path=path) # --------------------------------------------------------------------------- # Observation Normaliser # --------------------------------------------------------------------------- class ObservationNormalizer: """ Wraps RunningMeanStd. Call .normalize(obs) at every step. Statistics persist across sessions via .save()/.load(). """ def __init__(self, obs_dim: int, momentum: float = 0.001, stats_path: str = "obs_norm_stats.npz") -> None: self.rms = RunningMeanStd((obs_dim,), momentum=momentum) self.stats_path = stats_path self.rms.load(stats_path) def normalize(self, obs: np.ndarray, update: bool = True) -> np.ndarray: if update: self.rms.update(obs) return np.clip(self.rms.normalize(obs), -10.0, 10.0) def save(self) -> None: self.rms.save(self.stats_path) # --------------------------------------------------------------------------- # Observation Builder # --------------------------------------------------------------------------- class ObservationBuilder: """ Assembles the flat observation vector from all sensor sources. The set of included observations is configurable via YAML under the `observation_components` key. #!/usr/bin/env python3 """ VoltronAI v3.0 — PolicyStack Quad-redundancy: 1. High-level ONNX/SB3 policy (discrete state selection) 2. Low-level ONNX/SB3 policy (joint velocity deltas) 3. Shadow PD controller (runs in parallel) 4. If RL output deviates > threshold from PD → fall back to PD TensorRT path for NVIDIA Jetson Orin. Model versioning + SHA256 checksum verification. """ from __future__ import annotations import hashlib import os from pathlib import Path from typing import Dict, Optional, Tuple import numpy as np import structlog log = structlog.get_logger(__name__) # --------------------------------------------------------------------------- # Model loader with checksum # --------------------------------------------------------------------------- def _load_onnx_session(path: str, use_tensorrt: bool = False): """Load ONNX Runtime InferenceSession with optional TensorRT EP.""" import onnxruntime as ort if use_tensorrt: providers = [ ("TensorrtExecutionProvider", { "trt_max_workspace_size": 1 << 30, "trt_fp16_enable": True, }), "CUDAExecutionProvider", "CPUExecutionProvider", ] elif os.getenv("VOLTRON_CUDA", "0") == "1": providers = ["CUDAExecutionProvider", "CPUExecutionProvider"] else: providers = ["CPUExecutionProvider"] sess = ort.InferenceSession(path, providers=providers) log.info("onnx_session_loaded", path=path, providers=str(providers)) return sess class PolicyRunner: """ Wraps an ONNX session OR a stable-baselines3 PPO model. Verifies SHA256 on load when checksum is provided. """ def __init__( self, sb3_path: str, onnx_path: str, use_onnx: bool = True, use_tensorrt: bool = False, expected_sha256: Optional[str] = None, ) -> None: self.use_onnx = use_onnx self._sess = None self._model = None if use_onnx and Path(onnx_path).exists(): if expected_sha256: sha = hashlib.sha256(Path(onnx_path).read_bytes()).hexdigest() if sha != expected_sha256: raise RuntimeError( f"ONNX checksum mismatch for {onnx_path}: " f"expected {expected_sha256}, got {sha}" ) try: self._sess = _load_onnx_session(onnx_path, use_tensorrt) self._input_name = self._sess.get_inputs()[0].name return except Exception as exc: log.warning("onnx_load_failed_fallback", error=str(exc)) # Fallback: SB3 self.use_onnx = False if Path(sb3_path).exists(): from stable_baselines3 import PPO self._model = PPO.load(sb3_path, device="cpu") log.info("sb3_model_loaded", path=sb3_path) else: log.warning("no_model_found", sb3=sb3_path, onnx=onnx_path) def predict(self, obs: np.ndarray, deterministic: bool = True) -> np.ndarray: if self._sess is not None: inp = obs[np.newaxis, :].astype(np.float32) result = self._sess.run(None, {self._input_name: inp}) return result[0][0] if self._model is not None: action, _ = self._model.predict(obs, deterministic=deterministic) return action return np.zeros_like(obs[:1]) # safe no-op # --------------------------------------------------------------------------- # Shadow PD Controller # --------------------------------------------------------------------------- class ShadowPDController: """ Simple impedance-like PD controller that runs in parallel with the RL policy. If RL joint targets deviate more than `threshold` rad from PD targets in the worst joint, the PD output is used instead. """ def __init__(self, config, threshold: float = 0.3) -> None: n = len(config.joint_names) self.kp = np.full(n, 50.0) self.kd = np.full(n, 5.0) self.nominal = np.zeros(n) # upright nominal pose self.threshold = threshold self._dt = 1.0 / config.control_hz def compute(self, joint_pos: np.ndarray, joint_vel: np.ndarray) -> np.ndarray: """Return desired joint position from PD law (1-step ahead).""" torque = self.kp * (self.nominal - joint_pos) - self.kd * joint_vel delta_pos = torque * self._dt / self.kp # approximate return joint_pos + delta_pos def should_fallback(self, rl_target: np.ndarray, pd_target: np.ndarray) -> bool: deviation = float(np.max(np.abs(rl_target - pd_target))) if deviation > self.threshold: log.warning("shadow_pd_fallback_triggered", deviation=deviation) return True return False # --------------------------------------------------------------------------- # PolicyStack # --------------------------------------------------------------------------- class PolicyStack: """ Combines high-level + low-level policies with shadow PD fallback. Returns (state_index, joint_target_pos). """ def __init__(self, config) -> None: chk = config.model_checksums self.high_level = PolicyRunner( config.high_level_sb3, config.high_level_onnx, use_onnx=config.use_onnx, use_tensorrt=config.use_tensorrt, expected_sha256=chk.get(config.high_level_onnx), ) self.low_level = PolicyRunner( config.low_level_sb3, config.low_level_onnx, use_onnx=config.use_onnx, use_tensorrt=config.use_tensorrt, expected_sha256=chk.get(config.low_level_onnx), ) self.shadow_pd = ShadowPDController( config, threshold=config.shadow_pd_deviation_threshold ) self._max_joint_vel = config.max_joint_vel self._dt = 1.0 / config.control_hz self._n = len(config.joint_names) def infer( self, obs: np.ndarray, joint_pos: np.ndarray, joint_vel: np.ndarray, thermal_factor: float = 1.0, ) -> Tuple[int, np.ndarray]: """ Returns: state_idx — integer index into STATE_MAP target_pos — joint position target (safe) """ # High-level: discrete action try: raw_high = self.high_level.predict(obs, deterministic=False) state_idx = int(np.clip( int(np.rint(float(raw_high.item() if raw_high.ndim > 0 else raw_high))), 0, 11 # len(State) - 1 )) except Exception as exc: log.error("high_level_predict_failed", error=str(exc)) state_idx = 0 # IDLE # Low-level: velocity deltas → position targets try: raw_low = self.low_level.predict(obs, deterministic=True) delta_q = (np.array(raw_low, dtype=np.float32) * self._max_joint_vel * thermal_factor) rl_target = joint_pos + delta_q * self._dt except Exception as exc: log.error("low_level_predict_failed", error=str(exc)) rl_target = joint_pos.copy() # Shadow PD (always computed) pd_target = self.shadow_pd.compute(joint_pos, joint_vel) # Quad-redundancy: fall back to PD if RL deviates excessively if self.shadow_pd.should_fallback(rl_target, pd_target): target_pos = pd_target else: target_pos = rl_target return state_idx, target_pos

#!/usr/bin/env python3 """ VoltronAI v3.0 — CentroidalMPC Primary path: CasADi + acados (much faster than cvxpy on CPU). Fallback: cvxpy + OSQP (v2.5 behaviour). Runs in a daemon thread at 20 Hz. Exposes `last_solution` for the main control loop to query. """ ROBOT_MASS = 85.0 # kg GRAVITY = 9.81 # m/s² def __init__(self, config) -> None: self.config = config self.last_solution: Optional[float] = None self._lock = threading.Lock() self._state_ref: Optional[np.ndarray] = None self._ref_traj: Optional[np.ndarray] = None if not MPC_AVAILABLE: log.warning("mpc_disabled_no_solver") return if _CASADI_OK: self._build_casadi(config) else: self._build_cvxpy(config) self._thread = threading.Thread( target=self._solver_loop, daemon=True, name="mpc_solver" ) self._thread.start() log.info("mpc_started", backend="casadi" if _CASADI_OK else "cvxpy") # ----------------------------------------------------------------------- def _build_casadi(self, config) -> None: """Build CasADi NLP for centroidal MPC (fast, compiled).""" N = config.mpc_horizon dt = config.mpc_dt m = self.ROBOT_MASS g = self.GRAVITY opti = ca.Opti() # Decision variables com = opti.variable(N, 3) # CoM position vel = opti.variable(N, 3) # CoM velocity grf = opti.variable(N, 12) # Ground reaction forces (4 feet × 3) # Parameters com0 = opti.parameter(3) vel0 = opti.parameter(3) com_ref = opti.parameter(N, 3) # Cost cost = 0 for t in range(N): cost += 10 * ca.sumsqr(com[t, :] - com_ref[t, :]) cost += ca.sumsqr(vel[t, :]) cost += 1e-3 * ca.sumsqr(grf[t, :]) # Dynamics constraints cons = [] for t in range(N - 1): total_f = ca.sum2(ca.reshape(grf[t, :], (4, 3))) # sum over 4 feet acc = total_f / m - ca.DM([0, 0, g]).T cons.append(com[t+1, :] == com[t, :] + vel[t, :] * dt) cons.append(vel[t+1, :] == vel[t, :] + acc * dt) # Initial condition cons.append(com[0, :] == com0.T) cons.append(vel[0, :] == vel0.T) # Friction cone constraints (simplified) mu = 0.7 for t in range(N): for i in range(4): fz = grf[t, 3*i+2] fx = grf[t, 3*i] fy = grf[t, 3*i+1] cons.append(ca.fabs(fx) <= mu * fz) cons.append(ca.fabs(fy) <= mu * fz) cons.append(fz >= 50) cons.append(fz <= 1500) for c in cons: opti.subject_to(c) opti.minimize(cost) opti.solver("ipopt", {"ipopt.print_level": 0, "print_time": 0}) self._opti = opti self._com_var = com self._vel_var = vel self._grf_var = grf self._com0 = com0 self._vel0 = vel0 self._com_ref = com_ref def _build_cvxpy(self, config) -> None: """Fallback cvxpy + OSQP builder (v2.5 behaviour).""" import cvxpy as cp N, dt, m, g = ( config.mpc_horizon, config.mpc_dt, self.ROBOT_MASS, self.GRAVITY, ) self._cp_com = cp.Parameter(3) self._cp_vel = cp.Parameter(3) self._cp_com_ref = cp.Parameter((N, 3)) com_v = cp.Variable((N, 3)) vel_v = cp.Variable((N, 3)) grf = cp.Variable((N, 12)) cost, cons = 0, [] for t in range(N): cost += 10 * cp.sum_squares(com_v[t] - self._cp_com_ref[t]) cost += cp.sum_squares(vel_v[t]) cost += 1e-3 * cp.sum_squares(grf[t]) if t > 0: total_f = cp.sum(grf[t].reshape((4, 3)), axis=0) cons += [ com_v[t] == com_v[t-1] + vel_v[t-1] * dt, vel_v[t] == vel_v[t-1] + (total_f / m - np.array([0, 0, g])) * dt, ] for i in range(4): fz = grf[t, 3*i+2]; fx = grf[t, 3*i]; fy = grf[t, 3*i+1] cons += [cp.abs(fx) <= 0.7*fz, cp.abs(fy) <= 0.7*fz, fz >= 50, fz <= 1500] self._cp_prob = cp.Problem(cp.Minimize(cost), cons) self._cp_com_v = com_v self._cp_vel_v = vel_v # ----------------------------------------------------------------------- def request_solve(self, state: np.ndarray, ref_traj: np.ndarray) -> None: """Non-blocking: stash state/ref for the solver thread.""" with self._lock: self._state_ref = state.copy() self._ref_traj = ref_traj.copy() def _solver_loop(self) -> None: while True: try: with self._lock: state = self._state_ref ref = self._ref_traj if state is not None and ref is not None: if _CASADI_OK: self._solve_casadi(state, ref) else: self._solve_cvxpy(state, ref) except Exception as exc: log.warning("mpc_solver_error", error=str(exc)) time.sleep(1.0 / 20.0) def _solve_casadi(self, state: np.ndarray, ref: np.ndarray) -> None: try: self._opti.set_value(self._com0, state[:3]) self._opti.set_value(self._vel0, state[3:6]) self._opti.set_value(self._com_ref, ref) sol = self._opti.solve() with self._lock: self.last_solution = float(sol.value(self._com_var[0, 2])) except Exception as exc: log.warning("casadi_solve_failed", error=str(exc)) def _solve_cvxpy(self, state: np.ndarray, ref: np.ndarray) -> None: import cvxpy as cp try: self._cp_com.value = state[:3] self._cp_vel.value = state[3:6] self._cp_com_ref.value = ref self._cp_prob.solve(solver=cp.OSQP, warm_start=True) with self._lock: self.last_solution = ( self._cp_prob.value if self._cp_prob.status == cp.OPTIMAL else None ) except Exception as exc: log.warning("cvxpy_solve_failed", error=str(exc))

#!/usr/bin/env python3 """ VoltronAI v3.0 — SensorFusion + GraspDetector Fixes: - LiDAR: proper angular binning (no [::2] slice assumption) - GraspDetector: MobileNetV3-based CNN via ONNX (replaces pure heuristics) """ from __future__ import annotations from pathlib import Path from typing import Optional import numpy as np import structlog log = structlog.get_logger(__name__) try: import rospy from sensor_msgs.msg import LaserScan, Image from std_msgs.msg import Float32MultiArray from cv_bridge import CvBridge _ROS_AVAILABLE = True except ImportError: _ROS_AVAILABLE = False # --------------------------------------------------------------------------- # SensorFusion: LiDAR + human detection # --------------------------------------------------------------------------- class SensorFusion: """ Handles LiDAR scan processing with proper angular binning. Bins are equal-angle spans; each bin returns the minimum range (most conservative / closest obstacle). Falls back to 5.0 m if no scan received. """ if self._raw_ranges is None or self._raw_angles is None: return np.full(self.num_bins, 5.0, dtype=np.float32) ranges = self._raw_ranges angles = self._raw_angles bins = np.full(self.num_bins, 5.0, dtype=np.float32) a_min = float(angles.min()) a_max = float(angles.max()) bin_width = (a_max - a_min) / self.num_bins for i in range(self.num_bins): lo = a_min + i * bin_width hi = lo + bin_width mask = (angles >= lo) & (angles < hi) & (ranges > 0.05) if np.any(mask): bins[i] = float(np.min(ranges[mask])) return bins def inject_sim_scan(self, ranges: np.ndarray, angles: np.ndarray) -> None: """Simulation mode: inject scan directly.""" self._raw_ranges = ranges.astype(np.float32) self._raw_angles = angles.astype(np.float32) # --------------------------------------------------------------------------- # GraspDetector: CNN-based (MobileNetV3 via ONNX) # --------------------------------------------------------------------------- class GraspDetector: """ Fuses wrist F/T, gripper pressure, and camera to produce grasp probability. v3: uses a lightweight MobileNetV3 ONNX model for image-based scoring instead of the simple heuristic `image_ok` flag. """ def __init__(self, config) -> None: self._last_ft: Optional[np.ndarray] = None self._pressure: float = 0.0 self._image: Optional[np.ndarray] = None self._cnn_sess = None # Load CNN model cnn_path = config.grasp_cnn_model if Path(cnn_path).exists(): try: import onnxruntime as ort self._cnn_sess = ort.InferenceSession( cnn_path, providers=["CPUExecutionProvider"] ) self._cnn_input = self._cnn_sess.get_inputs()[0].name log.info("grasp_cnn_loaded", path=cnn_path) except Exception as exc: log.warning("grasp_cnn_load_failed", error=str(exc)) else: log.warning("grasp_cnn_not_found", path=cnn_path, note="falling back to heuristic") if _ROS_AVAILABLE and not config.sim_mode: self._bridge = CvBridge() rospy.Subscriber(config.wrist_ft_topic, Float32MultiArray, self._ft_cb) rospy.Subscriber(config.gripper_pressure_topic, Float32MultiArray, self._pressure_cb) rospy.Subscriber(config.camera_topics[0], Image, self._image_cb) def _ft_cb(self, msg) -> None: self._last_ft = np.array(msg.data[:6], dtype=np.float32) def _pressure_cb(self, msg) -> None: self._pressure = float(msg.data[0]) if msg.data else 0.0 def _image_cb(self, msg) -> None: try: self._image = self._bridge.imgmsg_to_cv2(msg, "bgr8") except Exception: pass def _image_score(self) -> float: """Run CNN on current image patch, return score in [0, 1].""" if self._image is None: return 0.0 if self._cnn_sess is not None: try: import cv2 img = cv2.resize(self._image, (224, 224)).astype(np.float32) / 255.0 img = np.transpose(img, (2, 0, 1))[np.newaxis] # NCHW result = self._cnn_sess.run(None, {self._cnn_input: img}) # Assume binary classifier output: sigmoid score return float(1.0 / (1.0 + np.exp(-result[0][0, 1]))) except Exception as exc: log.warning("grasp_cnn_inference_failed", error=str(exc)) # Fallback heuristic: image simply present → 0.5 return 0.5 def detect_probability(self) -> float: """Return grasp success probability in [0, 1].""" if self._last_ft is None: return 0.0 ft_mag = float(np.linalg.norm(self._last_ft[:3])) pressure_ok = min(self._pressure / 90.0, 1.0) ft_ok = min(ft_mag / 36.0, 1.0) image_score = self._image_score() return float(pressure_ok * 0.35 + ft_ok * 0.35 + image_score * 0.30) def detect_success(self) -> bool: return self.detect_probability() >= 0.5

#!/usr/bin/env python3 """ VoltronAI v3.0 — TeleopHandler Velocity-ramped teleop: commands are smoothed over a ramp window to prevent sudden jerks when the operator sends step inputs. """ from __future__ import annotations import time from typing import Optional import numpy as np import structlog log = structlog.get_logger(__name__) try: import rospy from std_msgs.msg import Bool from sensor_msgs.msg import JointState _ROS_AVAILABLE = True except ImportError: _ROS_AVAILABLE = False class TeleopHandler: """ Subscribes to /teleop/enable and /teleop/joint_cmd. Applies velocity ramping so operator step inputs are smoothed. """ RAMP_TIME = 0.2 # seconds to reach commanded velocity def __init__(self, config, actuation) -> None: self.config = config self.actuation = actuation self.active = False self._n = len(config.joint_names) self._target = np.zeros(self._n) self._current = np.zeros(self._n) self._last_t = time.time() if _ROS_AVAILABLE and not config.sim_mode: rospy.Subscriber("/teleop/enable", Bool, self._enable_cb) rospy.Subscriber("/teleop/joint_cmd", JointState, self._joint_cb) def _enable_cb(self, msg) -> None: self.active = msg.data if not self.active: # Ramp back to safe pose self._target = np.zeros(self._n) log.info("teleop_enabled" if self.active else "teleop_disabled") def _joint_cb(self, msg) -> None: if not self.active: return for i, name in enumerate(msg.name): for j, jn in enumerate(self.config.joint_names): if name == jn and i < len(msg.position): self._target[j] = msg.position[i] def step(self, joint_pos: np.ndarray) -> np.ndarray: """ Compute ramped command. Returns the smoothed position target. Call this from the main teleop loop. """ now = time.time() dt = now - self._last_t self._last_t = now alpha = min(dt / self.RAMP_TIME, 1.0) self._current = (1 - alpha) * self._current + alpha * self._target # Safety clamp pos_lim = np.array(self.config.joint_pos_limits[:self._n]) cmd = np.clip(self._current, -pos_lim, pos_lim) self.actuation.publish(cmd, joint_pos, duration=0.05) return cmd

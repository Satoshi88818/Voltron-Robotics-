Overview & Architecture {#overview}

VoltronAI is a full-stack humanoid robot controller designed for 85 kg real-hardware deployment. It combines:

UKF-based leg-aided state estimation (contact zero-velocity updates)

Hierarchical RL (high-level discrete policy + low-level velocity PPO)

Centroidal MPC (OSQP-backed, runs in a background thread at 20 Hz)

Real sensor integration (LiDAR, IMU, F/T, gripper pressure, camera)

Triple-redundant safety (watchdog, zone enforcement, hardware limits)

ROS1 (with ROS2/rclpy migration path)

STATE ESTIMATION — Leg-Aided UKF with Contact Zero-Velocity Update # ============================================================================= class StateEstimator: """ 15-dimensional UKF: x = [px, py, pz, (0:3) vx, vy, vz, (3:6) roll, pitch, yaw, (6:9) ωx, ωy, ωz, (9:12) abx, aby, abz] (12:15) ← accelerometer bias Leg-aided extension: When at least one foot is in contact with the ground the world-frame velocity is clamped to zero for that support phase (zero-velocity update). """ def __init__(self, config: Dict): self.dt = 1.0 / config["state_estimator_hz"] n_joints = len(config["joint_names"]) self.ukf = UnscentedKalmanFilter( dim_x=15, dim_z=18, dt=self.dt, fx=self._dynamics, hx=self._measurement ) self.ukf.x[:9] = np.zeros(9) self.ukf.P *= 0.1 self.ukf.R = np.eye(18) * 0.05 self.ukf.Q = Q_discrete_white_noise(3, dt=self.dt, var=0.02, block_size=5) self.last_imu: Optional[Imu] = None self.last_joints: Optional[JointState] = None self.contact: List[bool] = [False, False] # [left_foot, right_foot] self.joint_pos: np.ndarray = np.zeros(n_joints) self.joint_vel: np.ndarray = np.zeros(n_joints) # ← v2.5: real velocity self._name_to_idx: Dict[str, int] = { n: i for i, n in enumerate(config["joint_names"]) } rospy.Subscriber(config["imu_topic"], Imu, self._imu_cb) rospy.Subscriber(config["joint_state_topic"], JointState, self._joint_cb) # ----- ROS callbacks ----- def _imu_cb(self, msg: Imu) -> None: self.last_imu = msg def _joint_cb(self, msg: JointState) -> None: self.last_joints = msg pos = list(self.joint_pos) vel = list(self.joint_vel) for i, name in enumerate(msg.name): if name in self._name_to_idx: idx = self._name_to_idx[name] pos[idx] = msg.position[i] if i < len(msg.position) else 0.0 # v2.5: populate real joint velocities vel[idx] = msg.velocity[i] if i < len(msg.velocity) else 0.0 self.joint_pos = np.array(pos, dtype=np.float32) self.joint_vel = np.array(vel, dtype=np.float32) # ----- UKF functions ----- def _dynamics(self, x: np.ndarray, dt: float) -> np.ndarray: x = x.copy() if self.last_imu is None: return x r = Rotation.from_euler("xyz", x[6:9]) R = r.as_matrix() raw_acc = np.array([ self.last_imu.linear_acceleration.x, self.last_imu.linear_acceleration.y, self.last_imu.linear_acceleration.z, ]) acc = R @ (raw_acc - x[12:15]) # body → world, bias-corrected x[0:3] += x[3:6] * dt + 0.5 * acc * dt * dt x[3:6] += acc * dt x[6:9] += x[9:12] * dt return x def _measurement(self, x: np.ndarray) -> np.ndarray: z = np.zeros(18) z[0:12] = x[0:12] if self.last_imu is not None: z[12:15] = [ self.last_imu.angular_velocity.x, self.last_imu.angular_velocity.y, self.last_imu.angular_velocity.z, ] z[15:18] = [ self.last_imu.linear_acceleration.x, self.last_imu.linear_acceleration.y, self.last_imu.linear_acceleration.z, ] return z # ----- Main update (called every control tick) ----- def update(self) -> np.ndarray: if self.last_imu is None or self.last_joints is None: return np.zeros(15) self.ukf.predict() # v2.5: contact zero-velocity update (leg-aided) if self.contact[0] or self.contact[1]: self.ukf.x[3:6] = 0.0 # clamp world-frame velocity while foot planted z = self._measurement(self.ukf.x) self.ukf.update(z) return self.ukf.x.copy() # ============================================================================= # 2. REAL GRASP DETECTION — now returns float probability (v2.5) # ============================================================================= class GraspDetector: """ Fuses wrist F/T, gripper pressure, and camera to produce a grasp success *probability* (0.0–1.0). Threshold at 0.5 for bool usage. """ def __init__(self): self.bridge = CvBridge() self.last_ft = None self.pressure = 0.0 self.image = None rospy.Subscriber(CONFIG["wrist_ft_topic"], Float32MultiArray, self._ft_cb) rospy.Subscriber(CONFIG["gripper_pressure_topic"], Float32MultiArray, self._pressure_cb) rospy.Subscriber(CONFIG["camera_topics"][0], Image, self._image_cb) def _ft_cb(self, msg: Float32MultiArray) -> None: self.last_ft = np.array(msg.data[:6], dtype=np.float32) def _pressure_cb(self, msg: Float32MultiArray) -> None: self.pressure = float(msg.data[0]) if msg.data else 0.0 def _image_cb(self, msg: Image) -> None: try: self.image = self.bridge.imgmsg_to_cv2(msg, "bgr8") except Exception: pass def detect_probability(self) -> float: """Return grasp success probability in [0, 1].""" if self.last_ft is None: return 0.0 ft_mag = np.linalg.norm(self.last_ft[:3]) pressure_ok = min(self.pressure / 90.0, 1.0) # 45 Pa threshold → 0.5 ft_ok = min(ft_mag / 36.0, 1.0) # 18 N threshold → 0.5 image_ok = 1.0 if self.image is not None else 0.0 return float(pressure_ok * 0.4 + ft_ok * 0.4 + image_ok * 0.2) def detect_success(self) -> bool: return self.detect_probability() >= 0.5 # ============================================================================= # 3. CENTROIDAL MPC — Background thread at 20 Hz (v2.5) # ============================================================================= try: import cvxpy as cp MPC_AVAILABLE = True except ImportError: MPC_AVAILABLE = False logger.warning("cvxpy not installed → MPC fallback disabled") class CentroidalMPC: """ Centroidal dynamics MPC with OSQP backend. v2.5: runs in a daemon thread at 20 Hz so it never blocks the 100 Hz control loop. The latest solution is always available via .last_solution. """ def __init__(self, config: Dict): self.last_solution: Optional[float] = None self._lock = threading.Lock() self._state_ref: Optional[np.ndarray] = None self._ref_traj: Optional[np.ndarray] = None if not MPC_AVAILABLE: return N, dt, m, g = ( config["mpc_horizon"], config["mpc_dt"], 85.0, # robot mass kg 9.81, ) self.com = cp.Parameter(3) self.com_vel = cp.Parameter(3) self.com_ref = cp.Parameter((N, 3)) com_var = cp.Variable((N, 3)) vel_var = cp.Variable((N, 3)) grf = cp.Variable((N, 12)) cost, cons = 0, [] for t in range(N): cost += 10 * cp.sum_squares(com_var[t] - self.com_ref[t]) cost += cp.sum_squares(vel_var[t]) cost += 0.001 * cp.sum_squares(grf[t]) if t > 0: cons += [ com_var[t] == com_var[t-1] + vel_var[t-1] * dt, vel_var[t] == vel_var[t-1] + ( cp.sum(grf[t].reshape((4, 3)), axis=0) / m - np.array([0, 0, g]) ) * dt, ] for i in range(4): fz = grf[t, 3*i+2] fx = grf[t, 3*i] fy = grf[t, 3*i+1] cons += [ cp.abs(fx) <= 0.7 * fz, cp.abs(fy) <= 0.7 * fz, fz >= 50, fz <= 1500, ] self.prob = cp.Problem(cp.Minimize(cost), cons) # v2.5: background solver thread self._thread = threading.Thread(target=self._solver_loop, daemon=True) self._thread.start() logger.info("CentroidalMPC background thread started at 20 Hz") def request_solve(self, state: np.ndarray, ref_traj: np.ndarray) -> None: """Non-blocking: stash latest state/ref for the solver thread.""" with self._lock: self._state_ref = state.copy() self._ref_traj = ref_traj.copy() def _solver_loop(self) -> None: while True: with self._lock: state = self._state_ref ref = self._ref_traj if state is not None and ref is not None and MPC_AVAILABLE: try: self.com.value = state[:3] self.com_vel.value = state[3:6] self.com_ref.value = ref self.prob.solve(solver=cp.OSQP, warm_start=True) with self._lock: self.last_solution = ( self.prob.value if self.prob.status == cp.OPTIMAL else None ) except Exception as exc: logger.warning(f"MPC solver error: {exc}") time.sleep(1.0 / 20.0) # 20 Hz # ============================================================================= # 4. IK SAFETY LAYER — Joint-limit clamp + null-space projection (v2.5) # ============================================================================= class IKSafetyLayer: """ Lightweight safety filter applied to every low-level policy output. Clamps joint positions to hardware limits and (optionally) applies a null-space projection to keep the solution near a safe nominal pose. To use full pinocchio IK, install it and swap in the commented block. """ def __init__(self, config: Dict): n = len(config["joint_names"]) self.pos_limits = np.array(config["joint_pos_limits"][:n], dtype=np.float32) self.vel_limits = np.array(config["joint_vel_limits"][:n], dtype=np.float32) self.nominal = np.zeros(n, dtype=np.float32) def apply(self, target_pos: np.ndarray, current_pos: np.ndarray) -> np.ndarray: """ Clamp target_pos to [-pos_limits, +pos_limits]. Optionally clamp velocity: |Δq| / dt must not exceed vel_limits. """ # 1. Position limits safe = np.clip(target_pos, -self.pos_limits, self.pos_limits) # 2. Velocity limit (dt ≈ 0.01 s at 100 Hz) dt = 0.01 dq = safe - current_pos dq_max = self.vel_limits * dt dq = np.clip(dq, -dq_max, dq_max) safe = current_pos + dq return safe.astype(np.float32) # ============================================================================= # 5. ONNX INFERENCE WRAPPER (optional GPU path) — v2.5 # ============================================================================= class PolicyRunner: """ Wraps either a stable-baselines3 PPO model or an ONNX session. Set USE_ONNX=1 in environment to activate TensorRT/ONNX path. """ def __init__(self, sb3_path: str, onnx_path: str, device: str = "cpu"): self.use_onnx = bool(int(os.getenv("USE_ONNX", "0"))) if self.use_onnx: try: import onnxruntime as ort providers = ( ["CUDAExecutionProvider", "CPUExecutionProvider"] if device == "cuda" else ["CPUExecutionProvider"] ) self.sess = ort.InferenceSession(onnx_path, providers=providers) self.input_name = self.sess.get_inputs()[0].name logger.info(f"PolicyRunner: ONNX session loaded from {onnx_path}") except Exception as exc: logger.warning(f"ONNX load failed ({exc}), falling back to SB3") self.use_onnx = False if not self.use_onnx: from stable_baselines3 import PPO self.model = PPO.load(sb3_path, device=device) logger.info(f"PolicyRunner: SB3 model loaded from {sb3_path}") def predict(self, obs: np.ndarray, deterministic: bool = True) -> np.ndarray: if self.use_onnx: inp = obs[np.newaxis, :].astype(np.float32) result = self.sess.run(None, {self.input_name: inp}) return result[0][0] else: action, _ = self.model.predict(obs, deterministic=deterministic) return action # ============================================================================= # MAIN CLASS — VoltronAI v2.5 (Real Hardware) # ============================================================================= class VoltronAI: """ Full-stack hierarchical controller for a real 85 kg humanoid robot. 1)) command = self.state_map[action_idx] if self.state not in (State.EMERGENCY_STOP, State.SAFE_MODE): self.state = command except Exception as exc: logger.error(f"High-level policy predict failed: {exc}") command = State.IDLE # --- Low-level policy (joint velocity deltas) --- # v2.5: policy outputs Δq (velocities), NOT absolute positions try: raw_low = self.low_level_runner.predict(obs, deterministic=True) delta_q = np.array(raw_low, dtype=np.float32) * self.config["max_joint_vel"] target_pos = self.state_estimator.joint_pos + delta_q * 0.01 # integrate Δq * dt except Exception as exc: logger.error(f"Low-level policy predict failed: {exc}") target_pos = self.state_estimator.joint_pos.copy() # hold current pose # --- IK safety layer --- safe_pos = self.ik_safety.apply(target_pos, self.state_estimator.joint_pos) # --- Publish --- self.publish_joint_trajectory(safe_pos, duration=0.08) # --- MPC (non-blocking, 20 Hz background) --- if self.mpc is not None: ref_traj = np.tile(self.state_estimator.update()[:3], (self.config["mpc_horizon"], 1)) self.mpc.request_solve(self.state_estimator.update(), ref_traj) # --- Battery (real monitoring via /power/status; this is a safety floor check only) --- if self.battery_level <= 0: logger.critical("Battery depleted — ESTOP") self.emergency_stop() # ========================================================================= # Background Loops # ========================================================================= def _watchdog_loop(self) -> None: """v2.5: hardware watchdog — ESTOP if no command published within timeout.""" while True: timeout = self.config["watchdog_timeout"] if ( self.state not in (State.EMERGENCY_STOP, State.TELEOP, State.IDLE) and time.time() - self.last_command_time > timeout ): logger.critical(f"WATCHDOG TIMEOUT ({timeout}s) — ESTOP") self.emergency_stop() time.sleep(0.1) def _metrics_loop(self) -> None: while True: uptime_gauge.set(time.time() - self.start_time) battery_gauge.set( self.battery_level / self.config["battery_capacity"] * 100 ) state_gauge.labels(state=self.state.value).set(1) time.sleep(1.0) def _heartbeat_loop(self) -> None: while True: heartbeat_gauge.set(time.time()) time.sleep(0.5) # ============================================================================= # ENTRY POINT # ============================================================================= if __name__ == "__main__": rospy.init_node("voltron_ai_v25", anonymous=True) robot = VoltronAI() rate = rospy.Rate(CONFIG["control_hz"]) logger.info("VoltronAI v2.5 — READY FOR METAL. Dependencies

ONNX Inference (GPU acceleration)

Export your SB3 model to ONNX once, then set USE_ONNX=1:

# Export script (run once after training) import torch from stable_baselines3 import PPO model = PPO.load("models/high_level_policy.zip") obs_dim = model.observation_space.shape[0] dummy = torch.zeros(1, obs_dim) torch.onnx.export(model.policy, dummy, "models/high_level_policy.onnx", input_names=["obs"], output_names=["action"]) USE_ONNX=1 rosrun voltron_ai voltron_ai_v25.py
